                                                        这个例子：
                                                        const obj = {};

                                                        let foo = Symbol("foo");

                                                        Object.defineProperty(obj, foo, {
                                                        value: "foobar",
                                                        });

                                                        for (let i in obj) {
                                                        console.log(i); // 无输出
                                                        }

                                                        Object.getOwnPropertyNames(obj)
                                                        // []

                                                        Object.getOwnPropertySymbols(obj)

                                                        经过defineProperty定义的属性本身默认是enumerable false的，所以for...in本来就不会输出定义的属性，换成如下的例子能更好的说明symbol属性不会被遍历输出：

                                                        const obj = {};

                                                        let foo = Symbol("foo");

                                                        Object.defineProperty(obj, foo, {
                                                        value: "foobar",
                                                        enumerable: true
                                                        });

                                                        Object.defineProperty(obj, "enumerableAttr", {
                                                        value: "enumerableAttr",
                                                        enumerable: true
                                                        });

                                                        for (let i in obj) {
                                                        console.log(i); // 'enumerableAttr'
                                                        }

                                                        console.log(Object.getOwnPropertyNames(obj))
                                                        // [ "enumerableAttr" ]
